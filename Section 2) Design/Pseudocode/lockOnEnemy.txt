PROCEDURE isInRange(EnemyPosition, WeaponPosition, Radius)
	int dx = WeaponPosition.x - EnemyPosition.x
	int dy = WeaponPosition.y - EnemyPosition.y
	int squared = (dx^2) + (dy^2)
	int squareroot = squared^0.5
	IF (squareroot > Radius) THEN
		RETURN FALSE
	ELSE
		RETURN TRUE
	END IF
END PROCEDURE
